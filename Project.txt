//crindo banco de dados
CREATE DATABASE IF NOT EXISTS db_bigbike;
//selecionado a tabela
USE db_bigbike;

//criando tabela tb_classe_principal
CREATE TABLE IF NOT EXISTS tb_classe_principal (
  id int NOT NULL AUTO_INCREMENT,
  descricao varchar(50) COLLATE utf8_unicode_ci DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8mb4_0900_ai_ci;

//criando tabela tb_classe_secundaria
CREATE TABLE IF NOT EXISTS tb_classe_secundaria (
  id int NOT NULL AUTO_INCREMENT,
  descricao varchar(50) COLLATE utf8_unicode_ci DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8_unicode_ci;

//criando tabela tb_clientes
CREATE TABLE IF NOT EXISTS tb_clientes (
  id int NOT NULL AUTO_INCREMENT,
  nome varchar(75) CHARACTER SET utf8 COLLATE utf8_unicode_ci DEFAULT NULL,
  telefone varchar(13) COLLATE utf8_unicode_ci DEFAULT NULL,
  email varchar(75) COLLATE utf8_unicode_ci DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8_unicode_ci;

//criando tabela tb_funcionarios
CREATE TABLE IF NOT EXISTS tb_funcionarios (
  id int NOT NULL AUTO_INCREMENT,
  nome varchar(75) CHARACTER SET utf8 COLLATE utf8_unicode_ci DEFAULT NULL,
  data_nascimento date DEFAULT NULL,
  telefone varchar(13) CHARACTER SET utf8 COLLATE utf8_unicode_ci DEFAULT NULL,
  salario_base decimal(7,2) DEFAULT NULL,
  carga_horaria int DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8_unicode_ci;

//criando tabela tb_manutencoes
CREATE TABLE IF NOT EXISTS tb_manutencoes (
  id int NOT NULL AUTO_INCREMENT,
  id_cliente int DEFAULT NULL,
  id_funcionario int DEFAULT NULL,
  data_entrada date DEFAULT NULL,
  inicio_manutencao datetime DEFAULT NULL,
  fim_manutencao datetime DEFAULT NULL,
  data_prevista date DEFAULT NULL,
  data_entrega date DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8_unicode_ci;

//criando tabela tb_orcamentos
CREATE TABLE IF NOT EXISTS tb_orcamentos (
  id int NOT NULL AUTO_INCREMENT,
  id_cliente int DEFAULT NULL,
  id_manutencao int DEFAULT NULL,
  id_produto int DEFAULT NULL,
  qt_produto int DEFAULT NULL,
  id_servico int DEFAULT NULL,
  qt_servico int DEFAULT NULL,
  data_orcamento datetime DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8_unicode_ci;

//criando tabela tb_produtos
CREATE TABLE IF NOT EXISTS tb_produtos (
  id int NOT NULL AUTO_INCREMENT,
  id_classe_principal int DEFAULT NULL,
  id_classe_secundaria int DEFAULT NULL,
  descricao varchar(50) CHARACTER SET utf8 COLLATE utf8_unicode_ci DEFAULT NULL,
  vl_compra decimal(7,2) DEFAULT NULL,
  vl_venda decimal(7,2) DEFAULT NULL,
  qt_estoque int DEFAULT NULL,
  qt_min_estoque int DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8_unicode_ci;

//criando tabela tb_servicos
CREATE TABLE IF NOT EXISTS tb_servicos (
  id int NOT NULL AUTO_INCREMENT,
  descricao varchar(50) COLLATE utf8_unicode_ci DEFAULT NULL,
  vl_manutencao decimal(7,2) DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8_unicode_ci;

//criando tabela tb_vendas
CREATE TABLE IF NOT EXISTS tb_vendas (
  id int NOT NULL AUTO_INCREMENT,
  id_orcamento int DEFAULT NULL,
  data_venda date DEFAULT NULL,
  forma_pagamento enum('Dinheiro','Cheque') COLLATE utf8_unicode_ci DEFAULT NULL,
  parcelamento int DEFAULT NULL,
  PRIMARY KEY (id)
) COLLATE=utf8_unicode_ci;

//criando views
CREATE VIEW v_tempo_medio_valor_medio_man_valor_medio_horas_fun AS
SELECT SEC_TO_TIME(SUM(TIME_TO_SEC(a.fim_manutencao) - TIME_TO_SEC(a.inicio_manutencao))/COUNT(a.id)) tempo_medio_manutencao,
(SUM(b.qt_servico * c.vl_manutencao)) / COUNT(c.id) valor_medio_manutencao,
(d.salario_base / 30)/d.carga_horaria valor_medio_horas_funcionario
FROM tb_manutencoes a
JOIN tb_orcamentos b
ON b.id_cliente = a.id_cliente
JOIN tb_servicos c
ON c.id = b.id_servico
JOIN tb_funcionarios d
ON d.id = a.id_funcionario;

CREATE VIEW v_bicicletas_dia AS
SELECT data_entrada,
COUNT(data_entrada) qt_bicicletas
FROM tb_manutencoes
GROUP BY data_entrada;

CREATE VIEW v_custo_funcionario AS
SELECT nome,
(salario_base * 0.08) +  salario_base custo_fgts,
(salario_base / 12) * 0.08 + (salario_base / 12) prov_13salario,
(salario_base / 3 / 12 * 0.08) + (salario_base / 3 / 12) prov_ferias,
(salario_base * 0.08) +  salario_base + (salario_base / 12 + 0.08) + (salario_base / 3 / 12 + 0.08) costo_mensal
FROM tb_funcionarios;

CREATE VIEW v_fora_do_prazo AS
SELECT COUNT(id) fora_do_prazo
from tb_manutencoes
WHERE data_prevista < data_entrega;

CREATE VIEW v_lucro_geral AS
SELECT b.descricao,
b.vl_compra,
b.vl_venda,
b.vl_venda - b.vl_compra lucro_pecas
FROM tb_orcamentos a
left JOIN tb_produtos b
ON b.id = a.id_produto
JOIN tb_classe_principal c
ON c.id = b.id_classe_principal;

CREATE VIEW v_lucro_pecas AS
SELECT b.descricao,
b.vl_compra,
b.vl_venda,
b.vl_venda - b.vl_compra lucro_pecas
FROM tb_orcamentos a
left JOIN tb_produtos b
ON b.id = a.id_produto
JOIN tb_classe_principal c
ON c.id = b.id_classe_principal
WHERE b.id_classe_principal <> 2
AND b.id_classe_principal <> 3
AND b.id_classe_principal <> 4
AND b.id_classe_principal <> 6
AND b.id_classe_principal <> 8;

CREATE VIEW v_tempo_medio_manutencao AS
SELECT SEC_TO_TIME(SUM(TIME_TO_SEC(fim_manutencao) - TIME_TO_SEC(inicio_manutencao))/COUNT(id)) tempo_medio_manutencao
FROM tb_manutencoes;

CREATE VIEW v_valor_caixa_ano AS
SELECT a.data_venda,
SUM(b.qt_produto * c.vl_venda) vl_total_venda,
SUM(d.vl_manutencao * b.qt_servico) vl_total_servico,
(SUM(b.qt_produto * c.vl_venda) + COALESCE(SUM((d.vl_manutencao * b.qt_servico)),0)) total_caixa_dia
FROM tb_vendas a
JOIN tb_orcamentos b
ON b.id = a.id_orcamento
LEFT JOIN tb_produtos c
ON c.id = b.id_produto
LEFT JOIN tb_servicos d
ON d.id = b.id_servico
WHERE forma_pagamento = 'Dinheiro'
GROUP BY YEAR(data_venda);

CREATE VIEW v_valor_caixa_dia AS
SELECT a.data_venda,
SUM(b.qt_produto * c.vl_venda) vl_total_venda,
SUM(d.vl_manutencao * b.qt_servico) vl_total_servico,
(SUM(b.qt_produto * c.vl_venda) + COALESCE(SUM((d.vl_manutencao * b.qt_servico)),0)) total_caixa_dia
FROM tb_vendas a
JOIN tb_orcamentos b
ON b.id = a.id_orcamento
LEFT JOIN tb_produtos c
ON c.id = b.id_produto
LEFT JOIN tb_servicos d
ON d.id = b.id_servico
WHERE forma_pagamento = 'Dinheiro'
GROUP BY data_venda;

CREATE VIEW v_valor_caixa_mes AS

SUM(d.vl_manutencao * b.qt_servico) vl_total_servico,
(SUM(b.qt_produto * c.vl_venda) + COALESCE(SUM((d.vl_manutencao * b.qt_servico)),0)) total_caixa_dia
FROM tb_vendas a

ON c.id = b.id_produto

CREATE VIEW v_valor_entrada_setores AS
SELECT SUM(b.qt_produto * c.vl_venda) vl_total_venda,
SUM(d.vl_manutencao * b.qt_servico) vl_total_servico,
(SUM(b.qt_produto * c.vl_venda) + SUM((d.vl_manutencao * b.qt_servico))) total
FROM tb_vendas a
JOIN tb_orcamentos b
ON b.id = a.id_orcamento
LEFT JOIN tb_produtos c
ON c.id = b.id_produto
LEFT JOIN tb_servicos d
ON d.id = b.id_servico;

CREATE VIEW v_valor_medio_manutencao AS
SELECT SUM(d.vl_manutencao) / COUNT(b.qt_servico) vl_medio_manutencao
FROM tb_vendas a
JOIN tb_orcamentos b
ON b.id = a.id_orcamento
LEFT JOIN tb_produtos c
ON c.id = b.id_produto
LEFT JOIN tb_servicos d
ON d.id = b.id_servico
WHERE forma_pagamento = 'Dinheiro'
AND b.id_manutencao IS NOT NULL;
